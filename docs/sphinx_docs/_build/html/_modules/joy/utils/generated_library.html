[docs]def rest(stack):
  """
  ::

    ([a1 ...0] -- [...0])

  """
  try:
    s0, stack = stack
  except ValueError:
    raise StackUnderflowError
  if not isinstance(s0, tuple):
    raise NotAListError('Not a list.')
  try:
    _, s1 = s0
  except ValueError:
    raise StackUnderflowError('Cannot take rest of empty list.')
  return (s1, stack)
